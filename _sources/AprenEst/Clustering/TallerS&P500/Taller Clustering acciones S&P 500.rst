.. code:: ipython3

    # Obtener la lista de tickers del S&P 500 desde Wikipedia
    url = "https://en.wikipedia.org/wiki/List_of_S%26P_500_companies"
    table = pd.read_html(url)
    tickers_df = table[0]

    # Descargar datos históricos para cada ticker
    data = {}
